    canvas {

      background: radial-gradient(circle at center, #111 0%, #000 100%);

      border: 3px solid #0ff;

      border-radius: 12px;

      box-shadow: 0 0 20px #0ff;

      display: block;

      margin-bottom: 15px;

    }


    #message {

      font-size: 1.3em;

      text-align: center;

      color: #0ff;

      text-shadow: 0 0 5px #0ff, 0 0 10px #0ff;

      min-height: 2em;

      transition: opacity 0.5s;

    }


    #score {

      font-size: 1.2em;

      margin-bottom: 10px;

<body>

  <div id="score">Score: 0</div>

  <canvas id="gameCanvas" width="400" height="400"></canvas>

  <div id="message"></div>


  <script>

    const canvas = document.getElementById("gameCanvas");

    const ctx = canvas.getContext("2d");

    const box = 20;

    const canvasSize = 400;


    let snake = [{ x: 8 * box, y: 8 * box }];

    let direction = "RIGHT";

    let food = spawnFood();

    let score = 0;

    let gameOver = false;


    const scoreElement = document.getElementById("score");

    const messageElement = document.getElementById("message");


    function spawnFood() {

      let x = Math.floor(Math.random() * (canvasSize / box)) * box;

      let y = Math.floor(Math.random() * (canvasSize / box)) * box;

      while (snake.some(segment => segment.x === x && segment.y === y)) {

        x = Math.floor(Math.random() * (canvasSize / box)) * box;

        y = Math.floor(Math.random() * (canvasSize / box)) * box;

      }

      return { x, y, glow: 0 };

    }


    function draw() {

      ctx.fillStyle = "#000";

      ctx.fillRect(0, 0, canvasSize, canvasSize);


      // Schlange zeichnen mit Neon-Effekt

      for (let i = 0; i < snake.length; i++) {

        const segment = snake[i];

        const gradient = ctx.createRadialGradient(

          segment.x + box/2, segment.y + box/2, box/4,

          segment.x + box/2, segment.y + box/2, box

        );

        gradient.addColorStop(0, i === 0 ? "#0ff" : "#0a0");

        gradient.addColorStop(1, "transparent");

        ctx.fillStyle = gradient;

        ctx.fillRect(segment.x, segment.y, box, box);

      }


      // Essen zeichnen mit Puls-Effekt

      const pulse = 0.2 * Math.sin(Date.now() / 150) + 1;

      const foodGradient = ctx.createRadialGradient(

        food.x + box/2, food.y + box/2, box/4,

        food.x + box/2, food.y + box/2, box/1.5

      );

      foodGradient.addColorStop(0, `rgba(255,0,0,${pulse})`);

      foodGradient.addColorStop(1, "rgba(255,0,0,0)");

      ctx.fillStyle = foodGradient;

      ctx.fillRect(food.x, food.y, box, box);

    }


    function update() {

      if (gameOver) return;


      let head = { ...snake[0] };


      switch (direction) {

        case "LEFT": head.x -= box; break;

        case "UP": head.y -= box; break;

        case "RIGHT": head.x += box; break;

        case "DOWN": head.y += box; break;

      }


      // Kollision mit Wänden

      if (head.x < 0 || head.x >= canvasSize || head.y < 0 || head.y >= canvasSize) {

        endGame();

        return;

      }


      // Kollision mit sich selbst

      if (snake.some(segment => segment.x === head.x && segment.y === head.y)) {

        endGame();

        return;

      }


      snake.unshift(head);


      // Essen essen

      if (head.x === food.x && head.y === food.y) {

        score++;

        scoreElement.textContent = `Score: ${score}`;

        if (score >= 20) {

          endGame(true);

          return;

        }

        food = spawnFood();

      } else {

        snake.pop();

      }


      draw();

    }


    function endGame(victory = false) {

      gameOver = true;

      if (victory) {

        messageElement.textContent = "Wir sind bald am Ziel angekommen! Vielleicht solltest du dir aber erst die Hände waschen!";

      } else {

        messageElement.textContent = "Game Over! Bitte starte neu.";

      }

      messageElement.style.opacity = "1";

      setTimeout(resetGame, 2500);

    }


    function resetGame() {

      snake = [{ x: 8 * box, y: 8 * box }];

      direction = "RIGHT";

      food = spawnFood();

      score = 0;

      gameOver = false;

      scoreElement.textContent = `Score: ${score}`;

      messageElement.style.opacity = "0";

      setTimeout(() => { messageElement.textContent = ""; }, 500);

    }


    document.addEventListener("keydown", e => {

      if (e.key === "ArrowLeft" && direction !== "RIGHT") direction = "LEFT";

      if (e.key === "ArrowUp" && direction !== "DOWN") direction = "UP";

      if (e.key === "ArrowRight" && direction !== "LEFT") direction = "RIGHT";

      if (e.key === "ArrowDown" && direction !== "UP") direction = "DOWN";

    });


    // Touchsteuerung für Handys

    let touchStartX = 0;

    let touchStartY = 0;


    canvas.addEventListener("touchstart", e => {

      touchStartX = e.touches[0].clientX;

      touchStartY = e.touches[0].clientY;

    });


    canvas.addEventListener("touchend", e => {

      const touchEndX = e.changedTouches[0].clientX;

      const touchEndY = e.changedTouches[0].clientY;


      const dx = touchEndX - touchStartX;

      const dy = touchEndY - touchStartY;


      if (Math.abs(dx) > Math.abs(dy)) {

        if (dx > 0 && direction !== "LEFT") direction = "RIGHT";

        if (dx < 0 && direction !== "RIGHT") direction = "LEFT";

      } else {

        if (dy > 0 && direction !== "UP") direction = "DOWN";

        if (dy < 0 && direction !== "DOWN") direction = "UP";

      }

    });


    setInterval(update, 130);

    draw();

  </script>
